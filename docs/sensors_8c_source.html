82bool throttle_in_callback(mcpwm_cap_channel_handle_t cap_chan,
83 const mcpwm_capture_event_data_t *edata,
84 void *user_ctx)
85{
86 static uint32_t pwm_pos_edge_ticks = 0;
87 if(edata->cap_edge == MCPWM_CAP_EDGE_POS)
88 {
89 pwm_pos_edge_ticks = edata->cap_value;
90 }
91 else // MCPWM_CAP_EDGE_NEG
92 {
93 taskENTER_CRITICAL_ISR(&throttle_in_spinlock);
94 throttle_in_duty_ticks = edata->cap_value - pwm_pos_edge_ticks;
95 taskEXIT_CRITICAL_ISR(&throttle_in_spinlock);
96 }
97 return true;
98}

226float get_velocity(void)
227{
228 taskENTER_CRITICAL(&tachometer_spinlock);
229 float rot_s = tachometer_counts /

235float get_throttle_in_duty(void)
236{
237 taskENTER_CRITICAL(&throttle_in_spinlock);
238 uint32_t out = throttle_in_duty_ticks;
239 taskEXIT_CRITICAL(&throttle_in_spinlock);
240 return out * 100.0*PWM_FREQ / esp_clk_apb_freq();
241}
242float get_distance(void)
243{
244 taskENTER_CRITICAL(&echo_spinlock);
245 uint32_t out = echo_tof_ticks;
246 taskEXIT_CRITICAL(&echo_spinlock);
247 return out * (343.0/2 / esp_clk_apb_freq());
248}
