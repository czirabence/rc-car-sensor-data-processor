81bool throttle_in_callback(mcpwm_cap_channel_handle_t cap_chan,
82 const mcpwm_capture_event_data_t *edata,
83 void *user_ctx)
84{
85 static uint32_t pwm_pos_edge_ticks = 0;
86 if(edata->cap_edge == MCPWM_CAP_EDGE_POS)
87 {
88 pwm_pos_edge_ticks = edata->cap_value;
89 }
90 else // MCPWM_CAP_EDGE_NEG
91 {
92 taskENTER_CRITICAL_ISR(&throttle_in_spinlock);
93 throttle_in_duty_ticks = edata->cap_value - pwm_pos_edge_ticks;
94 taskEXIT_CRITICAL_ISR(&throttle_in_spinlock);
95 }
96 return true;
97}

234float get_throttle_in_duty(void)
235{
236 taskENTER_CRITICAL(&throttle_in_spinlock);
237 uint32_t out = throttle_in_duty_ticks;
238 taskEXIT_CRITICAL(&throttle_in_spinlock);
239 return out * 100.0*PWM_FREQ / esp_clk_apb_freq();

241float get_distance(void)
242{
243 taskENTER_CRITICAL(&echo_spinlock);
244 uint32_t out = echo_tof_ticks;
245 taskEXIT_CRITICAL(&echo_spinlock);
246 return out * (343.0/2 / esp_clk_apb_freq());
247}
