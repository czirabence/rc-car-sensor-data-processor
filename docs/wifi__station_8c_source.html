51/* FreeRTOS event group to signal when we are connected*/
52static EventGroupHandle_t s_wifi_event_group;
53
54/* The event group allows multiple bits for each event, but we only care about two events:
55 * - we are connected to the AP with an IP
56 * - we failed to connect after the maximum amount of retries */
57#define WIFI_CONNECTED_BIT BIT0
58#define WIFI_FAIL_BIT BIT1
59
60static const char *TAG = "wifi station";
61
62static int s_retry_num = 0;
63
64
65static void event_handler(void* arg, esp_event_base_t event_base,
66 int32_t event_id, void* event_data)
67{
68 if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_START) {
69 esp_wifi_connect();
70 } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_DISCONNECTED) {
71 if (s_retry_num < EXAMPLE_ESP_MAXIMUM_RETRY) {
72 esp_wifi_connect();
73 s_retry_num++;
74 ESP_LOGI(TAG, "retry to connect to the AP");
75 } else {
76 xEventGroupSetBits(s_wifi_event_group, WIFI_FAIL_BIT);
77 }
78 ESP_LOGI(TAG,"connect to the AP fail");
79 } else if (event_base == IP_EVENT && event_id == IP_EVENT_STA_GOT_IP) {
80 ip_event_got_ip_t* event = (ip_event_got_ip_t*) event_data;
81 ESP_LOGI(TAG, "got ip:" IPSTR, IP2STR(&event->ip_info.ip));
82 s_retry_num = 0;
83 xEventGroupSetBits(s_wifi_event_group, WIFI_CONNECTED_BIT);
84 }
85}

87void wifi_init_sta(void)
88{
89 s_wifi_event_group = xEventGroupCreate();
90
91 ESP_ERROR_CHECK(esp_netif_init());
92
93 esp_netif_create_default_wifi_sta();
94
95 wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
96 ESP_ERROR_CHECK(esp_wifi_init(&cfg));
97
98 esp_event_handler_instance_t instance_any_id;
99 esp_event_handler_instance_t instance_got_ip;
100 ESP_ERROR_CHECK(esp_event_handler_instance_register(WIFI_EVENT,
101 ESP_EVENT_ANY_ID,
102 &event_handler,
103 NULL,
104 &instance_any_id));
105 ESP_ERROR_CHECK(esp_event_handler_instance_register(IP_EVENT,
106 IP_EVENT_STA_GOT_IP,
107 &event_handler,
108 NULL,
109 &instance_got_ip));
110
111 wifi_config_t wifi_config = {
112 .sta = {
113 .ssid = EXAMPLE_ESP_WIFI_SSID,
114 .password = EXAMPLE_ESP_WIFI_PASS,
115 /* Authmode threshold resets to WPA2 as default if password matches WPA2 standards (pasword len => 8).
116 * If you want to connect the device to deprecated WEP/WPA networks, Please set the threshold value
117 * to WIFI_AUTH_WEP/WIFI_AUTH_WPA_PSK and set the password with length and format matching to
118 * WIFI_AUTH_WEP/WIFI_AUTH_WPA_PSK standards.
119 */
120 .threshold.authmode = ESP_WIFI_SCAN_AUTH_MODE_THRESHOLD,
121 .sae_pwe_h2e = WPA3_SAE_PWE_BOTH,
122 },
123 };
124 ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_STA) );
125 ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_STA, &wifi_config) );
126 ESP_ERROR_CHECK(esp_wifi_start() );
127
128 ESP_LOGI(TAG, "wifi_init_sta finished.");
129
130 /* Waiting until either the connection is established (WIFI_CONNECTED_BIT) or connection failed for the maximum
131 * number of re-tries (WIFI_FAIL_BIT). The bits are set by event_handler() (see above) */
132 EventBits_t bits = xEventGroupWaitBits(s_wifi_event_group,
133 WIFI_CONNECTED_BIT | WIFI_FAIL_BIT,
134 pdFALSE,
135 pdFALSE,
136 portMAX_DELAY);
137
138 /* xEventGroupWaitBits() returns the bits before the call returned, hence we can test which event actually
139 * happened. */
140 if (bits & WIFI_CONNECTED_BIT) {
141 ESP_LOGI(TAG, "connected to ap SSID:%s password:%s",
142 EXAMPLE_ESP_WIFI_SSID, EXAMPLE_ESP_WIFI_PASS);
143 } else if (bits & WIFI_FAIL_BIT) {
144 ESP_LOGI(TAG, "Failed to connect to SSID:%s, password:%s",
145 EXAMPLE_ESP_WIFI_SSID, EXAMPLE_ESP_WIFI_PASS);
146 } else {
147 ESP_LOGE(TAG, "UNEXPECTED EVENT");
148 }
149}

151void nvs_init(void)
152{
153 esp_err_t ret = nvs_flash_init();
154 if (ret == ESP_ERR_NVS_NO_FREE_PAGES || ret == ESP_ERR_NVS_NEW_VERSION_FOUND) {
155 ESP_ERROR_CHECK(nvs_flash_erase());
156 ret = nvs_flash_init();

158 ESP_ERROR_CHECK(ret);
159
160 ESP_LOGI(TAG, "ESP_WIFI_MODE_STA");
161}
